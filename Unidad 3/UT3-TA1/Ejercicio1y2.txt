public static int enRango (int[] a, int bajo, int alto) {
	int contador = 0;					O(1)
	for (int i=0; i<a.length; i++) {			0(1)+O(n)
		if (a[i] >= bajo && a[i] < alto)		[O(1)+O(1)]
			contador++;				O(1)
	}
	return contador; O(1)
}
